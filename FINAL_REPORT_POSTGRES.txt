import asyncio
from archon.container import configure, get_repository_async

async def main():
    configure(repository_type="postgres")
    repo = await get_repository_async()  # Important: async version!

    # Use repository...
    total = await repo.count()
    print(f"Total pages: {total}")

    await repo.close()

asyncio.run(main())
